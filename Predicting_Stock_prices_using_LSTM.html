def build_model(train):
    model= Sequential()
    # add the first layer
    model.add(LSTM(units=50, return_sequences=True, input_shape=(train.shape[1], 1))) # units means neurons
    model.add(LSTM(units=50, return_sequences=True))
    model.add(LSTM(units=50, return_sequences=False))
    model.add(Dense(units=25)) # densly connected Neural network layer with 25 neurons
    model.add(Dense(units=1)) # densly connected NN with one layer
    
    # compile the model
    model.compile(optimizer='adam', loss='mean_squared_error', metrics=['accuracy'])
    
    return model
